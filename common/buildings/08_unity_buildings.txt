				OR = {
					is_gestalt = yes

		category = planet_buildings

			trigger = { exists = owner }

		}
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @ost_b1_upkeep
			minerals = @ost_b1_upkeep
		}
		produces = {
			unity = @ost_b1_produces
		}
		produces = {
			trigger = {
				exists = owner
				owner = {
					has_active_tradition = tr_benevolence_1
				}
			}
			unity = 5
		}
	}

	planet_modifier = {
		planet_jobs_unity_produces_mult = 0.05
		job_culture_worker_add = @b1_jobs
	}

	triggered_desc = {
		text = building_autochthon_monument_effect
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_solidarity_5
			}
		}
		modifier = {

	triggered_planet_modifier = {

		}
		modifier = {
			planet_districts_cities_ecocontinuity_upkeep_add = @ecocontinuity_upkeep_t1
		}
	}

	prerequisites = {
		"tech_cultural_heritage"
	}

	upgrades = {
		building_heritage_site
	}

	convert_to = {
